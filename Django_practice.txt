--views.py
from .models import cricketer
def show(request):
	search=request.GET.get("se")
	if search:
		data=cricketer.objects.filter(name__icontains=search)
		return render(request,"1.html",{"data":data})
	data=cricketer.objects.all()
	return render(request,"1.html",{"data":data})
